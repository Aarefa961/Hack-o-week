      </div>
    `).join('');
    updateStats();
  }

  function tagColor(tag) {
    return {Adventure:'#dde8e0',Beach:'#d9eaf5',City:'#e8e0d5',Culture:'#f0dfd0',Food:'#faebd7',Nature:'#dde8d8'}[tag] || '#e8dfd0';
  }

  function formatDate(d) {
    if (!d) return '';
    const dt = new Date(d);
    return dt.toLocaleDateString('en-US',{month:'short',day:'numeric',year:'numeric'});
  }

  function escHtml(s) {
    if(!s) return '';
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
  }

    const el = document.getElementById('tripSlots');
    el.innerHTML = TRIP_THEMES.map((t,i) => `

  function selectEmoji(i, e) {
    document.querySelectorAll(`#emojiPicker${i} .emoji-opt`).forEach(el=>el.classList.remove('selected'));
    document.querySelector(`#emojiPicker${i} [data-e="${e}"]`).classList.add('selected');
  }

  function getSelectedEmoji(i) {
    const s = document.querySelector(`#emojiPicker${i} .emoji-opt.selected`);
    return s ? s.getAttribute('data-e') : EMOJIS[i % EMOJIS.length];
  }

  function updateCount(el, countId, max) {
    document.getElementById(countId).textContent = el.value.length;
  }

  function openModal(type) {
    if (type === 'write') {
      buildTripSlots();
      document.getElementById('authorName').value = '';
      document.getElementById('blogTitle').value = '';
      document.getElementById('blogIntro').value = '';
      document.getElementById('introCount').textContent = '0';
      document.getElementById('writeModal').classList.add('open');
    } else if (type === 'view') {
      document.getElementById('viewModal').classList.add('open');
    }
    document.body.style.overflow = 'hidden';
  }

  function closeModal(type) {
    document.getElementById(type+'Modal').classList.remove('open');
    document.body.style.overflow = '';
  }

  document.getElementById('writeModal').addEventListener('click', function(e) {
    if (e.target === this) closeModal('write');
  });
  document.getElementById('viewModal').addEventListener('click', function(e) {
    if (e.target === this) closeModal('view');
  });

  function publishBlog() {
    const author = document.getElementById('authorName').value.trim();
    const title = document.getElementById('blogTitle').value.trim();
    const intro = document.getElementById('blogIntro').value.trim();
    const tag = document.getElementById('blogTag').value;

    if (!author) { showToast('Please enter your name ✍️'); return; }
    if (!title) { showToast('Please add a blog title 📝'); return; }

    const trips = [0,1,2].map(i => ({
      name: document.getElementById(`trip${i}name`).value.trim(),
      location: document.getElementById(`trip${i}loc`).value.trim(),
      story: document.getElementById(`trip${i}story`).value.trim(),
      emoji: getSelectedEmoji(i)
    }));

    const filledTrips = trips.filter(t => t.name && t.story);
    if (filledTrips.length < 1) {
      showToast('Add at least 1 trip story 🗺');
      return;
    }

    const blog = {
      id: Date.now(),
      author, title, intro, tag,
      emoji: filledTrips[0].emoji,
      trips: filledTrips,
      likes: 0, likedBy: [],
      date: new Date().toISOString().split('T')[0]
    };

    blogs.unshift(blog);
    save();
    renderGrid();
    closeModal('write');
    showToast('Your story is live! ✈️');
  }

  function viewBlog(id) {
    const b = blogs.find(x => x.id === id);
    if (!b) return;
    const sessionId = 'sess_' + Math.random().toString(36).slice(2);
    const liked = b.likedBy && b.likedBy.includes(sessionId);

    const content = document.getElementById('viewModalContent');
    content.innerHTML = `
      <button class="modal-close" onclick="closeModal('view')">✕</button>
      <div class="view-hero">${b.emoji}</div>
      <div class="card-tag" style="margin-bottom:0.75rem">${b.tag}</div>
      <div class="view-title">${escHtml(b.title)}</div>
      <div class="view-meta">
        <span>By <strong>${escHtml(b.author)}</strong></span>
        <span>${formatDate(b.date)}</span>
        <span>${b.trips.length} trip${b.trips.length>1?'s':''}</span>
      </div>
      ${b.intro ? `<div class="view-body">${escHtml(b.intro)}</div>` : ''}
      <div class="view-trips">
        ${b.trips.map((t,i) => `
          <div class="view-trip">
            <div class="view-trip-head">
              <span class="view-trip-emoji">${t.emoji}</span>
              <div>
                <div class="view-trip-name">${escHtml(t.name)}</div>
                ${t.location ? `<div class="view-trip-loc">📍 ${escHtml(t.location)}</div>` : ''}
              </div>
              <span style="margin-left:auto;font-size:0.75rem;color:var(--muted)">Trip ${i+1}</span>
            </div>
            <div class="view-trip-story">${escHtml(t.story)}</div>
          </div>
        `).join('')}
      </div>
      <div style="display:flex;align-items:center;gap:1rem;margin-top:1.5rem;padding-top:1rem;border-top:1px solid var(--border)">
        <button class="like-btn" id="likeBtn_${b.id}" onclick="likeBlog(${b.id})">
          ♥ <span id="likeCount_${b.id}">${b.likes}</span>
        </button>
        <span style="font-size:0.8rem;color:var(--muted)">Did this story inspire you?</span>
      </div>
    `;
    openModal('view');
  }

  function likeBlog(id) {
    const b = blogs.find(x => x.id === id);
    if (!b) return;
    b.likes++;
    save();
    const btn = document.getElementById(`likeBtn_${id}`);
    const count = document.getElementById(`likeCount_${id}`);
    if (btn) btn.classList.add('liked');
    if (count) count.textContent = b.likes;
    renderGrid();
    showToast('♥ Thanks for the love!');
  }
