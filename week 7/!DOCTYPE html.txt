    <script>
        // Set Date
        const options = { weekday: 'long', month: 'short', day: 'numeric' };
        document.getElementById('current-date').innerText = new Date().toLocaleDateString('en-US', options);

        // Habit Toggle Logic
        function toggleHabit(btn) {
            btn.classList.toggle('bg-indigo-600');
            btn.classList.toggle('text-white');
            btn.classList.toggle('border-indigo-600');
            saveData();
        }

        // Persistence Logic
        function saveData() {
            const data = {
                energy: document.getElementById('energy').value,
                notes: document.getElementById('notes').value
            };
            localStorage.setItem('pulse_data', JSON.stringify(data));
        }

        function resetDay() {
            if(confirm("Clear today's logs?")) {
                localStorage.removeItem('pulse_data');
                location.reload();
            }
        }

        // Auto-save on input
        document.querySelectorAll('input, textarea').forEach(el => {
            el.addEventListener('input', saveData);
        });
    </script>
