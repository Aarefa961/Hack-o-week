<script>
// ═══════════════════════════════════════════════════════
//  STATE
// ═══════════════════════════════════════════════════════
const state = {
  quiz: null,          // published quiz object
  participants: [],    // array of participant result objects
  sessionCode: null,
};

// ═══════════════════════════════════════════════════════
//  NAV
// ═══════════════════════════════════════════════════════
document.querySelectorAll('.nav-tab').forEach(tab => {
  tab.addEventListener('click', () => {
    document.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    tab.classList.add('active');
    document.getElementById(tab.dataset.screen).classList.add('active');
    if (tab.dataset.screen === 'dashScreen') renderDashboard();
  });
});

function switchTab(id) {
  document.querySelectorAll('.nav-tab').forEach(t => t.classList.toggle('active', t.dataset.screen === id));
  document.querySelectorAll('.screen').forEach(s => s.classList.toggle('active', s.id === id));
  if (id === 'dashScreen') renderDashboard();
}

// ═══════════════════════════════════════════════════════
//  BUILD QUESTION CARDS
// ═══════════════════════════════════════════════════════
const LETTERS = ['A','B','C','D'];
function buildQuestionCards() {
  const wrap = document.getElementById('questionsWrap');
  wrap.innerHTML = '';
  for (let i = 1; i <= 5; i++) {
    const div = document.createElement('div');
    div.className = 'card';
    div.innerHTML = `
      <div class="card-title"><span class="num">${i}</span> Question ${i}</div>
      <div class="form-group">
        <label>Question Text</label>
        <textarea id="q${i}_text" placeholder="Type your question here…"></textarea>
      </div>
      <div class="form-group">
        <label>Answer Choices</label>
        <div class="options-grid">
          ${LETTERS.map(l => `
            <div class="option-item">
              <span class="option-letter">${l}</span>
              <input type="text" id="q${i}_opt${l}" placeholder="Option ${l}" />
            </div>
          `).join('')}
        </div>
      </div>
      <div class="correct-radio">
        <label>Correct Answer:</label>
        <select class="correct-select" id="q${i}_correct">
          ${LETTERS.map(l => `<option value="${l}">${l}</option>`).join('')}
        </select>
      </div>
    `;
    wrap.appendChild(div);
  }
}
buildQuestionCards();

// ═══════════════════════════════════════════════════════
//  PUBLISH QUIZ
// ═══════════════════════════════════════════════════════
function publishQuiz() {
  const title = document.getElementById('quizTitle').value.trim();
  const subject = document.getElementById('quizSubject').value.trim();
  const teacher = document.getElementById('teacherName').value.trim();
  const timeLimit = parseInt(document.getElementById('quizTime').value) || 0;

  if (!title) { toast('Please enter a quiz title.'); return; }

  const questions = [];
  for (let i = 1; i <= 5; i++) {
    const text = document.getElementById(`q${i}_text`).value.trim();
    if (!text) { toast(`Question ${i} is empty.`); return; }
    const opts = {};
    for (const l of LETTERS) {
      const v = document.getElementById(`q${i}_opt${l}`).value.trim();
      if (!v) { toast(`Question ${i}: Option ${l} is empty.`); return; }
      opts[l] = v;
    }
    questions.push({ text, opts, correct: document.getElementById(`q${i}_correct`).value });
  }

  state.sessionCode = Math.random().toString(36).substr(2, 6).toUpperCase();
  state.quiz = { title, subject, teacher, timeLimit, questions, createdAt: new Date() };
  state.participants = [];

  toast('✅ Quiz published! Share the code with students.');
  setTimeout(() => switchTab('dashScreen'), 900);
}

function resetForm() {
  document.getElementById('quizTitle').value = '';
  document.getElementById('quizSubject').value = '';
  document.getElementById('quizTime').value = '10';
  document.getElementById('teacherName').value = '';
  buildQuestionCards();
  toast('Form reset.');
}

function renderDashboard() {
  const dc = document.getElementById('dashContent');
  const sub = document.getElementById('dashSub');

  if (!state.quiz) {
    sub.textContent = '// No quiz published yet.';
    dc.innerHTML = `<div class="empty"><div class="empty-icon">📋</div><p>Publish a quiz first to see participant data here.</p></div>`;
    return;
  }

  const q = state.quiz;
  const p = state.participants;
  sub.textContent = `// ${q.title} — ${q.subject || 'No subject'} · ${p.length} participant${p.length !== 1 ? 's' : ''}`;

  const done = p.filter(x => x.status === 'done');
  const avgScore = done.length ? (done.reduce((a,b) => a + b.score, 0) / done.length).toFixed(1) : '—';
  const perfect = done.filter(x => x.score === 5).length;
  const highPct = done.length ? Math.round((done.filter(x => x.score >= 4).length / done.length) * 100) : 0;

  // Per-question correct count
  const qStats = q.questions.map((_, qi) => {
    const correct = done.filter(px => px.answers && px.answers[qi] === q.questions[qi].correct).length;
    return done.length ? Math.round((correct / done.length) * 100) : 0;
  });

  dc.innerHTML = `
    <div class="quiz-meta">
      <div class="quiz-meta-item"><strong>${q.title}</strong></div>
      <div class="quiz-meta-sep">|</div>
      <div class="quiz-meta-item">Teacher: <strong>${q.teacher || '—'}</strong></div>
      <div class="quiz-meta-sep">|</div>

      <div class="quiz-meta-sep">|</div>

    </div>

    <div class="share-box">
      <div>
        <div class="share-label">Share this code with students</div>
        <div class="share-code">${state.sessionCode}</div>
      </div>
      <div style="text-align:right;">
        <button class="btn btn-secondary btn-sm" style="background:rgba(255,255,255,0.1);border-color:#555;color:#ccc;" onclick="copyCode()">📋 Copy Code</button>
        <div style="font-family:'DM Mono',monospace;font-size:0.75rem;color:#666;margin-top:0.5rem;">Students enter this in the "Student View" tab</div>
      </div>
    </div>

    <div class="stats-grid">
      <div class="stat-card">
        <div class="stat-val">${p.length}</div>
        <div class="stat-label">Total Joined</div>
      </div>
      <div class="stat-card">
        <div class="stat-val">${done.length}</div>
        <div class="stat-label">Completed</div>
      </div>
      <div class="stat-card">
        <div class="stat-val">${avgScore}</div>
        <div class="stat-label">Avg Score / 5</div>
      </div>
      <div class="stat-card">
        <div class="stat-val">${perfect}</div>
        <div class="stat-label">Perfect Scores</div>
      </div>
    </div>

    <!-- Per-question accuracy -->
    <div class="card" style="margin-bottom:1.5rem;">
      <div class="card-title">📊 Question Accuracy</div>
      ${q.questions.map((qq, qi) => `
        <div style="margin-bottom:0.85rem;">
          <div style="display:flex;justify-content:space-between;margin-bottom:0.3rem;font-size:0.82rem;">
            <span style="color:var(--slate);font-family:'DM Mono',monospace;">Q${qi+1}</span>
            <span style="font-weight:500;">${qStats[qi]}% correct</span>
          </div>
          <div style="background:var(--cream);border-radius:4px;height:8px;">
            <div style="width:${qStats[qi]}%;background:${qStats[qi]>=70?'var(--green)':qStats[qi]>=40?'var(--amber)':'var(--red)'};height:100%;border-radius:4px;transition:width 0.6s;"></div>
          </div>
        </div>
      `).join('')}
    </div>

    <!-- Participants Table -->
    <div class="card">
      <div class="card-title" style="justify-content:space-between;flex-wrap:wrap;gap:0.5rem;">

            </thead>
            <tbody>
              ${p.map((px, idx) => {
                const pct = px.status === 'done' ? Math.round((px.score/5)*100) : '—';
                const scoreClass = px.score >= 4 ? 'score-high' : px.score >= 2 ? 'score-mid' : 'score-low';
                const timeTaken = px.timeTaken ? formatSeconds(px.timeTaken) : '—';
                return `
                <tr>
                  <td style="font-family:'DM Mono',monospace;color:var(--slate);">${idx+1}</td>

function copyCode() {
  if (state.sessionCode) {
    navigator.clipboard.writeText(state.sessionCode).catch(() => {});
    toast('Code copied: ' + state.sessionCode);
  }
}

// ═══════════════════════════════════════════════════════
//  STUDENT VIEW: JOIN
// ═══════════════════════════════════════════════════════
function joinQuiz() {
  const code = document.getElementById('joinCode').value.trim().toUpperCase();
  const name = document.getElementById('joinName').value.trim();

  if (!code || code.length < 4) { toast('Enter a valid quiz code.'); return; }
  if (!name) { toast('Please enter your name.'); return; }

  if (!state.quiz || code !== state.sessionCode) {
    toast('❌ Quiz code not found. Ask your teacher!'); return;
  }

  const participant = {
    id: Date.now(),
    name,
    status: 'in-progress',
    score: 0,
    answers: [],
    joinedAt: new Date(),
    timeTaken: null,
  };
  state.participants.push(participant);

  document.getElementById('joinEntry').style.display = 'none';
  document.getElementById('quizRunner').style.display = 'block';
  startQuizRunner(participant);
}

// ═══════════════════════════════════════════════════════
//  STUDENT VIEW: QUIZ RUNNER
// ═══════════════════════════════════════════════════════
function startQuizRunner(participant) {
  const runner = document.getElementById('quizRunner');
  const q = state.quiz;
  let current = 0;
  const answers = [];
  const startTime = Date.now();
  let timerInterval = null;

  function renderQuestion() {
    const qi = current;
    const question = q.questions[qi];
    const pct = ((qi) / 5) * 100;

    runner.innerHTML = `
      <div class="quiz-container">
        <div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:0.75rem;">
          <div style="font-family:'DM Mono',monospace;font-size:0.8rem;color:var(--slate);">
            ${q.title}
          </div>
          ${q.timeLimit ? `<div style="font-family:'DM Mono',monospace;font-size:0.85rem;font-weight:500;" id="timerDisplay">⏱ ${q.timeLimit}:00</div>` : ''}
        </div>
        <div class="progress-bar-wrap">
          <div class="progress-bar-fill" style="width:${pct}%"></div>
        </div>
        <div class="q-card">
          <div class="q-num">Question ${qi + 1} of 5</div>
          <div class="q-text">${question.text}</div>
          <div class="q-options" id="optionsWrap">
            ${LETTERS.map(l => `
              <div class="q-option" data-letter="${l}" onclick="selectOption('${l}')">
                <span class="q-opt-letter">${l}</span>
                <span class="q-opt-text">${question.opts[l]}</span>
              </div>
            `).join('')}
          </div>
          <div class="btn-row" style="margin-top:1.5rem;justify-content:flex-end;">
            <button class="btn btn-primary" id="nextBtn" onclick="nextQuestion()" disabled>
              ${qi === 4 ? '✅ Submit Quiz' : 'Next →'}
            </button>
          </div>
        </div>
      </div>
    `;

    if (q.timeLimit && qi === 0) startTimer();
  }

  let selectedAnswer = null;

  window.selectOption = (letter) => {
    selectedAnswer = letter;
    document.querySelectorAll('.q-option').forEach(el => {
      el.classList.toggle('selected', el.dataset.letter === letter);
    });
    document.getElementById('nextBtn').disabled = false;
  };

  window.nextQuestion = () => {
    if (!selectedAnswer) return;
    answers.push(selectedAnswer);
    selectedAnswer = null;
    current++;
    if (current < 5) {
      renderQuestion();
    } else {
      finishQuiz();
    }
  };

  function startTimer() {
    let remaining = q.timeLimit * 60;
    timerInterval = setInterval(() => {
      remaining--;
      const m = Math.floor(remaining / 60);
      const s = remaining % 60;
      const el = document.getElementById('timerDisplay');
      if (el) el.textContent = `⏱ ${m}:${String(s).padStart(2,'0')}`;
      if (remaining <= 0) { clearInterval(timerInterval); finishQuiz(); }
    }, 1000);
  }

  function finishQuiz() {
    clearInterval(timerInterval);
    // Fill missing answers
    while (answers.length < 5) answers.push(null);
    const timeTaken = Math.round((Date.now() - startTime) / 1000);
    let score = 0;
    q.questions.forEach((qq, i) => { if (answers[i] === qq.correct) score++; });

    // Update participant
    const idx = state.participants.findIndex(p => p.id === participant.id);
    if (idx !== -1) {
      state.participants[idx].status = 'done';
      state.participants[idx].score = score;
      state.participants[idx].answers = answers;
      state.participants[idx].timeTaken = timeTaken;
    }

    // Show result
    const pct = Math.round((score / 5) * 100);
    const msg = score === 5 ? '🌟 Perfect Score!' : score >= 4 ? '🎉 Great Job!' : score >= 3 ? '👍 Good Effort!' : score >= 2 ? '📚 Keep Studying!' : '💪 Don\'t Give Up!';

    runner.innerHTML = `
      <div class="quiz-container">
        <div class="result-hero">
          <div class="result-score-ring">
            <div class="result-score-num">${score}</div>
            <div class="result-score-denom">out of 5</div>
          </div>
          <div class="result-title">${msg}</div>
          <div class="result-pct">${pct}% correct · ${formatSeconds(timeTaken)}</div>
        </div>

        <!-- Answer Review -->
        ${q.questions.map((qq, i) => {
          const chosen = answers[i];
          const isCorrect = chosen === qq.correct;
          return `
          <div class="card" style="margin-bottom:1rem;">
            <div style="display:flex;gap:0.75rem;align-items:flex-start;margin-bottom:0.75rem;">
              <span style="font-size:1.1rem;">${isCorrect ? '✅' : '❌'}</span>
              <span style="font-size:0.92rem;font-weight:500;">${qq.text}</span>
            </div>
            ${LETTERS.map(l => {
              let cls = '';
              if (l === qq.correct) cls = 'correct';
              else if (l === chosen && !isCorrect) cls = 'wrong';
              return `
              <div class="q-option ${cls}" style="cursor:default;margin-bottom:0.4rem;">
                <span class="q-opt-letter">${l}</span>
                <span class="q-opt-text">${qq.opts[l]}</span>
                ${l === qq.correct ? '<span style="margin-left:auto;font-size:0.75rem;color:var(--green);font-family:\'DM Mono\',monospace;">CORRECT</span>' : ''}
                ${l === chosen && !isCorrect ? '<span style="margin-left:auto;font-size:0.75rem;color:var(--red);font-family:\'DM Mono\',monospace;">YOUR ANSWER</span>' : ''}
              </div>`;
            }).join('')}
          </div>`;
        }).join('')}

        <div class="btn-row" style="justify-content:center;">
          <button class="btn btn-secondary" onclick="resetStudentView()">← Try Another Quiz</button>
        </div>
      </div>
    `;
  }

  renderQuestion();
}

window.resetStudentView = () => {
  document.getElementById('joinEntry').style.display = 'block';
  document.getElementById('quizRunner').style.display = 'none';
  document.getElementById('joinCode').value = '';
  document.getElementById('joinName').value = '';
};

// ═══════════════════════════════════════════════════════
//  UTILS
// ═══════════════════════════════════════════════════════
function formatTime(date) {
  if (!date) return '—';
  const d = new Date(date);
  return d.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
}
function formatSeconds(s) {
  if (!s) return '—';
  const m = Math.floor(s / 60);
  const sec = s % 60;
  return m > 0 ? `${m}m ${sec}s` : `${sec}s`;
}
function toast(msg) {
  const el = document.createElement('div');
  el.className = 'toast';
  el.textContent = msg;
  document.body.appendChild(el);
  setTimeout(() => el.remove(), 3000);
}
